### Part 1: Image Contours | 图像轮廓

轮廓是连接具有相同灰度值的连续点的曲线，是形状分析的基础。

#### 1.1 Find and Draw Contours | 查找与绘制轮廓

```python
import cv2
import matplotlib.pyplot as plt
import numpy as np

def cv_show(name, img):
    cv2.imshow(name, img)
    cv2.waitKey(0)
    cv2.destroyAllWindows()

# 读取图像并转为二值图
img = cv2.imread(r'D:\python-learn\python-YOLOv8\images\test1.jpg')
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
ret, thresh = cv2.threshold(gray, 127, 255, cv2.THRESH_BINARY)

# 查找轮廓
contours, hierarchy = cv2.findContours(thresh, cv2.RETR_TREE, cv2.CHAIN_APPROX_NONE)
# contours: 轮廓列表，每个轮廓是点集
# hierarchy: 层次结构信息

print(f'轮廓数量: {len(contours)}')

# 绘制所有轮廓（红色，线宽2）
draw_img = img.copy()
res = cv2.drawContours(draw_img, contours, -1, (0, 0, 255), 2)

# 绘制单个轮廓（下标为0）
draw_img2 = img.copy()
res2 = cv2.drawContours(draw_img2, contours, 0, (0, 0, 255), 2)
```

**参数说明：**
- `cv2.RETR_TREE`：检索所有轮廓并重建层次结构
- `cv2.CHAIN_APPROX_NONE`：保存所有轮廓点
- `cv2.CHAIN_APPROX_SIMPLE`：压缩水平/垂直/对角线方向，只保留端点

#### 1.2 Contour Features | 轮廓特征

```python
cnt = contours[23]  # 选取第23个轮廓

# 面积和周长
area = cv2.contourArea(cnt)
perimeter = cv2.arcLength(cnt, True)  # True表示闭合曲线
print(f'面积: {area}, 周长: {perimeter}')

# 轮廓近似（Douglas-Peucker算法）
epsilon = 0.1 * cv2.arcLength(cnt, True)  # 近似精度
approx = cv2.approxPolyDP(cnt, epsilon, True)

# 绘制近似轮廓
draw_img = img.copy()
res = cv2.drawContours(draw_img, [approx], -1, (0, 0, 255), 2)
```

#### 1.3 Bounding Rectangle & Circle | 边界矩形与圆

```python
# 边界矩形（外接矩形）
x, y, w, h = cv2.boundingRect(cnt)
img_rect = img.copy()
cv2.rectangle(img_rect, (x, y), (x + w, y + h), (0, 255, 0), 2)

# 轮廓面积与边界矩形面积比（Extent）
rect_area = w * h
extent = float(area) / rect_area
print(f'Extent: {extent}')

# 最小外接圆
(x, y), radius = cv2.minEnclosingCircle(cnt)
center = (int(x), int(y))
radius = int(radius)
img_circle = img.copy()
cv2.circle(img_circle, center, radius, (0, 255, 0), 2)
```

### Part 2: Template Matching | 模板匹配

模板匹配是在大图像中查找小模板位置的方法。

```python
img = cv2.imread('mario.jpg')
img_gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
template = cv2.imread('mario_coin.jpg', 0)
h, w = template.shape[:2]

# 多种匹配方法
methods = ['cv2.TM_CCOEFF', 'cv2.TM_CCOEFF_NORMED', 
           'cv2.TM_CCORR', 'cv2.TM_CCORR_NORMED', 
           'cv2.TM_SQDIFF', 'cv2.TM_SQDIFF_NORMED']

for meth in methods:
    img2 = img.copy()
    method = eval(meth)
    res = cv2.matchTemplate(img_gray, template, method)
    min_val, max_val, min_loc, max_loc = cv2.minMaxLoc(res)
    
    # SQDIFF方法取最小值位置，其他取最大值
    if method in [cv2.TM_SQDIFF, cv2.TM_SQDIFF_NORMED]:
        top_left = min_loc
    else:
        top_left = max_loc
    
    bottom_right = (top_left[0] + w, top_left[1] + h)
    cv2.rectangle(img2, top_left, bottom_right, (0, 255, 0), 2)
    
    plt.subplot(121), plt.imshow(res, cmap='gray')
    plt.subplot(122), plt.imshow(img2, cmap='gray')
    plt.show()

# 匹配多个对象（设置阈值）
threshold = 0.8
loc = np.where(res >= threshold)
for pt in zip(*loc[::-1]):  # 转换为(x, y)格式
    cv2.rectangle(img, pt, (pt[0] + w, pt[1] + h), (0, 255, 0), 2)
```

img = cv2.imread(r'D:\python-learn\python-YOLOv8\images\test1.jpg')

### Part 4: Histograms | 直方图

直方图用于分析图像的亮度分布，判断曝光是否正常。

#### 4.1 Basic Histogram | 基本直方图

```python
# 灰度图直方图
img = cv2.imread(r'D:\python-learn\python-YOLOv8\images\test1.jpg', 0)
hist = cv2.calcHist([img], [0], None, [256], [0, 256])  # 统计
plt.plot(hist)  # 折线图
plt.show()

plt.hist(img.ravel(), 256, [0, 256])  # 柱状图
plt.show()

# 彩色图直方图（BGR三个通道）
img_color = cv2.imread(r'D:\python-learn\python-YOLOv8\images\test1.jpg')
color = ('b', 'g', 'r')
for i, col in enumerate(color):
    histr = cv2.calcHist([img_color], [i], None, [256], [0, 256])
    plt.plot(histr, color=col)
plt.xlim([0, 256])
plt.show()
```

#### 4.2 Mask Application | 掩膜应用

```python
# 创建掩膜（只统计指定区域）
mask = np.zeros(img.shape[:2], np.uint8)
mask[100:300, 100:400] = 255

# 应用掩膜
masked_img = cv2.bitwise_and(img, img, mask=mask)

# 统计掩膜区域的直方图
hist_full = cv2.calcHist([img], [0], None, [256], [0, 256])
hist_mask = cv2.calcHist([img], [0], mask, [256], [0, 256])

plt.subplot(221), plt.imshow(img, 'gray')
plt.subplot(222), plt.imshow(mask, 'gray')
plt.subplot(223), plt.imshow(masked_img, 'gray')
plt.subplot(224), plt.plot(hist_full, 'b'), plt.plot(hist_mask, 'r')
plt.show()
```

#### 4.3 Histogram Equalization | 直方图均衡化

```python
# 全局均衡化（提高对比度）
equ = cv2.equalizeHist(img)
res = np.hstack((img, equ))

# 自适应均衡化（CLAHE，保留局部细节）
clahe = cv2.createCLAHE(clipLimit=2.0, tileGridSize=(8, 8))
res_clahe = clahe.apply(img)
res = np.hstack((img, equ, res_clahe))
cv_show('Global vs CLAHE', res)
```

### Part 5: Fourier Transform | 傅里叶变换

> "你眼中看似落叶纷飞变化无常的世界，实际只是躺在上帝怀中一份早已谱好的乐章。" —— 傅里叶变换让我们从频域角度理解图像

#### 5.1 Theory | 理论基础

- **高频**：变化剧烈的灰度分量（边缘、噪声）
- **低频**：变化缓慢的灰度分量（平滑区域）

| 滤波器 | 保留内容 | 效果 |
|--------|----------|------|
| 低通滤波 | 低频 | 图像模糊 |
| 高通滤波 | 高频 | 边缘增强 |

#### 5.2 Implementation | 代码实现

```python
img = cv2.imread(r'D:\python-learn\python-YOLOv8\images\test1.jpg', 0)
img_float32 = np.float32(img)  # DFT要求float32格式

# DFT变换
dft = cv2.dft(img_float32, flags=cv2.DFT_COMPLEX_OUTPUT)
dft_shift = np.fft.fftshift(dft)  # 低频移到中心

# 显示频谱图
magnitude_spectrum = 20 * np.log(cv2.magnitude(dft_shift[:, :, 0], dft_shift[:, :, 1]))

plt.subplot(121), plt.imshow(img, cmap='gray'), plt.title('Input')
plt.subplot(122), plt.imshow(magnitude_spectrum, cmap='gray'), plt.title('Spectrum')
plt.show()

# 创建掩膜进行滤波
rows, cols = img.shape
crow, ccol = rows // 2, cols // 2

# 低通滤波（保留中心60x60区域）
mask_low = np.zeros((rows, cols, 2), np.uint8)
mask_low[crow-30:crow+30, ccol-30:ccol+30] = 1

# 高通滤波（去除中心60x60区域）
mask_high = np.ones((rows, cols, 2), np.uint8)
mask_high[crow-30:crow+30, ccol-30:ccol+30] = 0

# 应用滤波并逆变换
fshift_low = dft_shift * mask_low
fshift_high = dft_shift * mask_high

f_ishift_low = np.fft.ifftshift(fshift_low)
f_ishift_high = np.fft.ifftshift(fshift_high)

img_low = cv2.idft(f_ishift_low)
img_high = cv2.idft(f_ishift_high)

img_low = cv2.magnitude(img_low[:, :, 0], img_low[:, :, 1])
img_high = cv2.magnitude(img_high[:, :, 0], img_high[:, :, 1])

# 展示结果
plt.subplot(221), plt.imshow(img, cmap='gray'), plt.title('Input Image')
plt.subplot(222), plt.imshow(magnitude_spectrum, cmap='gray'), plt.title('Magnitude Spectrum')
plt.subplot(223), plt.imshow(img_low, cmap='gray'), plt.title('low_fre back')
plt.subplot(224), plt.imshow(img_high, cmap='gray'), plt.title('high_fre back')
plt.show()
```

## 💡 Some Problems I Met... | 遇到的问题

### Problem 1: Contour Drawing Modifies Original Image | 轮廓绘制修改原图

- **What's wrong?** 直接在原图上绘制轮廓后，原图被永久修改。
- **Reason** `cv2.drawContours()` 会直接修改传入的图像。
- **Solution** 绘制前使用 `.copy()` 创建副本。

```python
# ✅ 正确做法
draw_img = img.copy()
res = cv2.drawContours(draw_img, contours, -1, (0, 0, 255), 2)
```

---

### Problem 2: Template Matching Coordinate Confusion | 模板匹配坐标混淆

- **What's wrong?** `np.where()` 返回的坐标顺序是 (y, x)，而绘图需要 (x, y)。
- **Reason** numpy的索引习惯是先行后列，而图像坐标系是先列后行。
- **Solution** 使用 `zip(*loc[::-1])` 进行坐标转换。

```python
# 坐标转换
for pt in zip(*loc[::-1]):  # (y, x) → (x, y)
    cv2.rectangle(img, pt, (pt[0]+w, pt[1]+h), (0, 255, 0), 2)
```

---

### Problem 3: DFT Output Not Displayable | DFT结果无法显示

- **What's wrong?** `cv2.dft()` 返回的复数结果直接显示是一片黑色。
- **Reason** DFT输出是复数（实部+虚部），幅度值范围极大（0到数百万），无法直接映射到0-255。
- **Solution** 计算幅度谱并取对数压缩动态范围。

```python
# 幅度计算 + 对数压缩
magnitude = cv2.magnitude(dft_shift[:, :, 0], dft_shift[:, :, 1])
magnitude_spectrum = 20 * np.log(magnitude + 1)  # +1避免log(0)
```

---

### Problem 4: Laplacian Pyramid Negative Values | 拉普拉斯金字塔出现负值

- **What's wrong?** `img - down_up` 结果出现负值，显示异常。
- **Reason** 图像相减可能产生负数，而图像显示需要0-255范围。
- **Solution** 使用 `cv2.normalize()` 归一化到0-255，或使用 `np.clip()` 截断。

```python
# 归一化处理
laplacian_norm = cv2.normalize(laplacian_pyr, None, 0, 255, cv2.NORM_MINMAX)
laplacian_norm = np.uint8(laplacian_norm)
```

```python
def get_contour_features(cnt):
    """提取轮廓的面积、周长、外接矩形、Extent"""
    area = cv2.contourArea(cnt)
    perimeter = cv2.arcLength(cnt, True)
    x, y, w, h = cv2.boundingRect(cnt)
    extent = area / (w * h)
    return {'area': area, 'perimeter': perimeter, 
            'bbox': (x, y, w, h), 'extent': extent}

def match_multiple_templates(img_gray, template, threshold=0.8):
    """匹配多个模板对象"""
    h, w = template.shape
    res = cv2.matchTemplate(img_gray, template, cv2.TM_CCOEFF_NORMED)
    loc = np.where(res >= threshold)
    matches = []
    for pt in zip(*loc[::-1]):
        matches.append((pt[0], pt[1], pt[0]+w, pt[1]+h))
    return matches

def apply_frequency_filter(img, filter_type='low', radius=30):
    """频域滤波（低通/高通）"""
    img_float = np.float32(img)
    dft = cv2.dft(img_float, flags=cv2.DFT_COMPLEX_OUTPUT)
    dft_shift = np.fft.fftshift(dft)
    
    rows, cols = img.shape
    crow, ccol = rows // 2, cols // 2
    
    if filter_type == 'low':
        mask = np.zeros((rows, cols, 2), np.uint8)
        mask[crow-radius:crow+radius, ccol-radius:ccol+radius] = 1
    else:  # high pass
        mask = np.ones((rows, cols, 2), np.uint8)
        mask[crow-radius:crow+radius, ccol-radius:ccol+radius] = 0
    
    fshift = dft_shift * mask
    f_ishift = np.fft.ifftshift(fshift)
    img_back = cv2.idft(f_ishift)
    result = cv2.magnitude(img_back[:, :, 0], img_back[:, :, 1])
    return np.uint8(np.clip(result, 0, 255))
```
