下面是我做的一个简单推理模型：
'
from ultralytics import YOLO
import os
model = YOLO('yolov8n.pt')
image_path = 'images/test.jpg'
results = model(image_path, save = True, show = False)
results[0].show()
print("Succeed!")
for result in results:
    boxes = result.boxes
    if boxes is not None:
        for box in boxes:
            cls_id = int(box.cls)
            conf = float(box.conf)
            print(f"检测到物体：{result.names[cls_id]}, 置信度：{conf:.2f}")
    else:
        print("未检测到任何物体。")
 '
